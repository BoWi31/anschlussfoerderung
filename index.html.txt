<body>
    <div id="status" class="status-container">
        <h1>Überprüfung läuft...</h1>
    </div>

    <script>
        // Förderzeiten und betroffene Schüler*innen
        const foerderzeiten = [
            { day: 1, start: "08:00", end: "09:30", schueler: ["Anastasia 5a", "Alina 5.1", "Marin 7.1", "Mohammed 7.1", "Ilias 7.2"] },
            { day: 5, start: "11:40", end: "13:10", schueler: ["Anastasia 5a", "Alina 5.1", "Marin 7.1", "Mohammed 7.1", "Ilias 7.2"] }
        ];

        function checkFoerderung() {
            const jetzt = new Date();
            const aktuelleZeit = `${jetzt.getHours().toString().padStart(2, "0")}:${jetzt.getMinutes().toString().padStart(2, "0")}`;
            const aktuellerTag = jetzt.getDay(); // Montag = 1, Sonntag = 0
            const statusDiv = document.getElementById("status");

            let foerderungAktiv = false;
            let aktuelleSchueler = [];

            // Zeit und Tag überprüfen
            foerderzeiten.forEach(zeit => {
                if (aktuellerTag === zeit.day && aktuelleZeit >= zeit.start && aktuelleZeit <= zeit.end) {
                    foerderungAktiv = true;
                    aktuelleSchueler = zeit.schueler;
                }
            });

            // Farbliche Anzeige je nach Status
            if (foerderungAktiv) {
                statusDiv.innerHTML = `
                    <h1>Anschlussförderung aktiv</h1>
                    <p>Betroffene Schüler*innen:</p>
                    <ul>
                        ${aktuelleSchueler.map(schueler => `<li>${schueler}</li>`).join("")}
                    </ul>`;
                statusDiv.className = "status-container red";
            } else {
                statusDiv.innerHTML = "<h1>Keine Anschlussförderung</h1>";
                statusDiv.className = "status-container green";
            }
        }

        // Überprüfung bei Seitenaufruf und alle 60 Sekunden
        setInterval(checkFoerderung, 60000);
        checkFoerderung();
    </script>
